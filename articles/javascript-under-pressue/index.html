    On a now defunct website, usvsth3m.com, a challenge game was created that tested the player's ability to write JavaScript quickly with a timer running. The game was aptly called "You Can't JavaScript Under Pressure". Although the original site has been shutdown, you can still find and check out the original site and game on the Internet Archive.

    // --2--
function isNumberEven(i) {
  // Again, nothing too crazy here.
  // All even numbers modulo 2 will have a zero remainder,
  // All odd numbers have a non-zero remainder
  return (i % 2) == 0    
}

    Question 3

    // --3--
// Things start to get interesting here ...
function getFileExtension(i) {
  /*
    The question generously grants us that the input will be in two formats:
    a.b
    -or-
    a

    So we just need to check when the format is "a.b" and return the 'b' part
  */

  // Let's split the string with a period into an array
  let s = i.split(".")

  // When we split the string, if there was a period, we would expect to have exactly two indexes in our array: [a,b]
  // If the array length isn't 2, we'll just return false.
  // If the array length is exactly 2, we return the second element
  return (s.length != 2) ? false : s[1]
}

    Question 4

    // --4--
function longestString(i) {
  // This is pretty straightforward...

  // Let's declare some temp vals to track the longest length and string we find
  // By default, the longest string we know about is a blank string with 0 length
  let len = 0
  let str = ""

  // Let's now work with the array we are given.
  // We should iterate over the entire array.
  for (let item of i){ // This code be read as saying "look at each item of the array and then do the following..."
    // We only care about looking for strings, so let's be sure the current item is a string
    if (typeof item === "string"){
      // We know this item is a string, so let's now see if this item is longer than the previous longest string
      if (item.length > len){
        // item in this case is longer, so let's update our temp vals before we jump back to the top of the loop
        len = item.length
        str = item 
      }
    }
  }
  // at the end of the full loop, we should be left with the longest string, which we can return
  return str
}

    Question 5

    // --5--
// This is a really fun question, albeit simple looking at first glance
function arraySum(i) {
  // Let's get our sum value going...
  let sum = 0

  // Let's now go through everything we have in the array handed to us...
  for (let item of i){
    // We know that each item in the array is going to be random.
    // We need to single out what the type each item is and figure out what to do with it
    
    // Numbers are the most imporant thing we are looking for.
    // Just add the number to our sum.
    if (typeof item === "number" )
      sum += item
    
    // If we find an array in our array, we should sum any of the numbers in it and add that to our sum
    else if (isArray(item)) // This isn't a builtin JavaScript method, see below...
      sum += arraySum(item)
    
    // Any other type doesn't matter to us, so we can jump back to the top of the loop
  }

  // The loop is done, so we have the complete sum to return
  return sum
}

/*
  As of the time of writing this article, JavaScript doesn't have a clear cut typeof statement for arrays.
  
  When you evaluate the expression `typeof []`, read as "what is type of an array literal" the result is "object".
  "object" isn't quite "array", but its different than most of the primitives that we can use it right?
  
  Not quite...
  `typeof {}`, read as "what is the type of an object literal", also results in "object".
  objects and arrays in JavaScript are somewhat similar, but they still behave very differently.
  
  So how can we work around this?
  Prototypes. The scary world and dark magics of JavaScript prototypes.

  This comment isn't the place to deep dive into the details of prototyping, but here's the important bits for us:
  - Everything has a prototype.
  - The prototype of a thing is roughly an analogue to the type of a thing
  - With prototypes, we can tell the difference between an object and an array

  So, if we want to know if a thing is an array, the following function will do just that:
*/
function isArray(a){
  return Object.prototype.toString.call(a) === '[object Array]'
}
